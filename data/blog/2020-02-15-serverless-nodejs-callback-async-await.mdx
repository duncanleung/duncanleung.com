```javascript
module.exports.handler = async (event, context, cb) => {
  const response = {
    statusCode: 200,
    body: JSON.stringify({ message: "hello world" })
  };

  cb(null, response);
};
```

```javascript
module.exports.handler = async (event, context) => {
  const response = {
    statusCode: 200,
    body: JSON.stringify({ message: "hello world" })
  };

  return response;
};
```
